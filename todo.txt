// Abstract box using a single-cell table
function abstractBox(lines) {
  const children = lines.flatMap((line, i) => {
    if (line === "") return [sp(40,40)];
    const parts = line.split(/(\*\*[^*]+\*\*)/g).map(seg => {
      if (seg.startsWith("**") && seg.endsWith("**"))
        return new TextRun({ text:seg.slice(2,-2), font:F, size:SZ_BODY, bold:true, color:INK });
      return new TextRun({ text:seg, font:F, size:SZ_BODY, color:INK });
    });
    return [ new Paragraph({ children:parts, spacing:{before:100, after:100, line:320} }) ];
  });
  const border = { style:BorderStyle.SINGLE, size:4, color:RULE };
  return new Table({
    width:{ size:9360, type:WidthType.DXA },
    columnWidths:[9360],
    rows:[ new TableRow({ children:[
      new TableCell({
        borders:{ top:border, bottom:border, left:border, right:border },
        shading:{ fill:BOXBG, type:ShadingType.CLEAR },
        margins:{ top:200, bottom:200, left:280, right:280 },
        width:{ size:9360, type:WidthType.DXA },
        children
      })
    ]}) ]
  });
}

  if (url) children.push(link(`↗`, url));
  return new Paragraph({
    children,
    spacing:{ before:80, after:80 },
    indent:{ left:400, hanging:400 }
  });
}
